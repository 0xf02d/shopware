class #classUc#RepositoryTest extends KernelTestCase
{
    /**
     * @var #classUc#Repository

    private $repository;

    {
        self::bootKernel();

    public function testSearchUuidsReturnsUuidSearchResult()
    {

        $result = $this->repository->search($criteria, $context);
        $this->assertInstanceOf(#classUc#SearchResult::class, $result);
    }
