        $builder->groupBy('#classLc#.#column#');

        $builder->select(['#classLc#.#column#', 'COUNT(#classLc#.uuid) as item_count']);

        $counts = $builder->execute()->fetchAll(\PDO::FETCH_KEY_PAIR);

        $true = 0;
        $false = 0;
        foreach($counts as $key => $count) {
            if ($key === 1) {
                $true = $count;
            } else {
                $false = $count;
            }
        }

        return new BooleanFacetResult($criteriaPart->getName(), $true, $false);
